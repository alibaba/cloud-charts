## en-US


````jsx
import { WLineG2 } from '@alife/aisc-widgets';

let options = {
  xAxis: {type: 'datetime'},
  keys: ["机房1", "机房2"]
};
class Demo extends React.Component{
  state = {
    data: (() => {
      let data = [
        {
          "name":"机房1",
          "data":[]
        },{
          "name":"机房2",
          "data":[]
        }
      ];

      const now = Date.now();
      for(let i = 0; i < 30; i++) {
        data[0].data.push([now - (30 - i) * 1000, Math.random() * 3000])
        data[1].data.push([now - (30 - i) * 1000, Math.random() * 3000])
      }

      return data;
    })()
  };

  componentDidMount() {
    setInterval(() => {
      const now = Date.now();
      this.state.data.forEach((oneData) => {
        oneData.data.shift();
        oneData.data.push([now, Math.random() * 3000]);
      });

      this.setState({
        data: [...this.state.data]
      })
    }, 1000);
  }

  render(){
    return (
      <div className="demos">
        <div className="demo-item">
            <WLineG2 height={300} configs={options} data={this.state.data}/>
        </div>
      </div>
    );
  }
}

ReactDOM.render(<Demo />, mountNode);
````
